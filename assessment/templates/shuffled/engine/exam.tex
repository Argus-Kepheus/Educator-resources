\newcommand{\inputExamSection}[1]{\input{sections/#1}}
\newcommand{\inputExamSectionsAuto}{%
  \directlua{
    local ok, lfs = pcall(require, "lfs")
    if not ok then
      tex.error("LuaFileSystem is not available", {"Set examSectionList manually, for example: S01,S02,S03."})
      return
    end
    local sections = {}
    for name in lfs.dir("sections") do
      if string.match(name, "^S[0-9][0-9]*[.]tex$") then
        table.insert(sections, name)
      end
    end
    table.sort(sections)
    if next(sections) == nil then
      tex.error("No exam sections found", {"Expected files such as sections/S01.tex."})
      return
    end
    for _, name in ipairs(sections) do
      local stem = string.sub(name, 1, -5)
      tex.print("\\input{sections/" .. stem .. "}")
    end
  }%
}
